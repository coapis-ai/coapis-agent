## 实施目标

### 主要目标

1. **自动检查必要配置文件**
   - 启动时检查关键配置文件是否存在
   - 输出友好的错误提示和修复建议

2. **自动创建默认配置**

```python
"""System diagnostics command."""
import click
from pathlib import Path
from typing import List, Tuple

@click.command()
@click.option('--fix', is_flag=True, help='Auto-fix issues')
def doctor(fix: bool):
    """Check system configuration and diagnose issues."""
    
    issues = []
    
    # 1. 检查数据目录
    issues.extend(check_data_directories())
    
    # 2. 检查配置文件
    issues.extend(check_config_files())
    
    # 3. 检查provider配置
    issues.extend(check_provider_configs())
    
    # 4. 检查场景数据
    issues.extend(check_scene_data())
    
    # 输出诊断报告
    print_diagnosis(issues)
    
    # 自动修复
    if fix:
        auto_fix(issues)

def check_config_files() -> List[Tuple[str, str, str]]:
    """检查配置文件"""
    issues = []
    
    required_files = [
        ("system/.secret/providers/active_model.json", "Provider active model config"),
        ("system/.secret/providers/default_models.json", "Default models config"),
        ("system/.secret/providers/hidden_providers.json", "Hidden providers config"),
        ("scenes.json", "Scene data"),
        ("tags.json", "Tag data"),
    ]
    
    for file_path, description in required_files:
        full_path = WORKING_DIR / file_path
        if not full_path.exists():
            issues.append(("missing_file", file_path, description))
    
    return issues

def check_provider_configs() -> List[Tuple[str, str, str]]:
    """检查provider配置"""
    issues = []
    
    custom_dir = WORKING_DIR / "system/.secret/providers/custom"
    if custom_dir.exists():
        providers = list(custom_dir.glob("*.json"))
        if not providers:
            issues.append(("no_provider", "custom/", "No provider configured"))
    
    return issues

def print_diagnosis(issues: List[Tuple[str, str, str]]):
    """输出诊断报告"""
    if not issues:
        click.echo(click.style("✅ System check passed!", fg="green", bold=True))
        return
    
    click.echo(click.style("═══════════════════════════════════════════════════════", fg="yellow"))
    click.echo(click.style("⚠️  System Configuration Issues Detected", fg="yellow", bold=True))
    click.echo(click.style("═══════════════════════════════════════════════════════", fg="yellow"))
    
    for issue_type, path, description in issues:
        if issue_type == "missing_file":
            click.echo(f"❌ Missing: {path}")
            click.echo(f"   Description: {description}")
        elif issue_type == "no_provider":
            click.echo(f"⚠️  Warning: {path}")
            click.echo(f"   {description}")
    
    click.echo("")
    click.echo(click.style("Quick Fix:", fg="cyan", bold=True))
    click.echo("  python -m coapis.cli.doctor --fix")
    click.echo("")
    click.echo(click.style("Manual Fix:", fg="cyan"))
    click.echo("  1. Configure provider: Settings → Providers")
    click.echo("  2. Or copy from template:")
    click.echo("     cp -r /template/system/.secret/providers/custom/*.json \\")
    click.echo("          /your/data/dir/system/.secret/providers/custom/")
    click.echo(click.style("═══════════════════════════════════════════════════════", fg="yellow"))

if __name__ == "__main__":
    doctor()
```

```python
"""System configuration check utilities."""
import logging
from pathlib import Path
from typing import List, Dict, Any

logger = logging.getLogger(__name__)

def check_system_config(data_dir: Path) -> Dict[str, Any]:
    """检查系统配置完整性
    
    Args:
        data_dir: 数据目录路径
    
    Returns:
        检查结果字典，包含：
        - is_ok: 是否通过检查
        - issues: 问题列表
        - warnings: 警告列表
    """
    issues = []
    warnings = []
    
    # 1. 检查必要配置文件
    required_files = {
        "system/.secret/providers/active_model.json": "Active model config",
        "system/.secret/providers/default_models.json": "Default models config",
        "system/.secret/providers/hidden_providers.json": "Hidden providers config",
    }
    
    for file_path, description in required_files.items():
        full_path = data_dir / file_path
        if not full_path.exists():
            issues.append({
                "type": "missing_file",
                "path": file_path,
                "description": description,
                "fix": f"Create default {file_path}"
            })
    
    # 2. 检查数据文件（警告级别）
    data_files = {
        "scenes.json": "Scene data",
        "tags.json": "Tag data",
    }
    
    for file_path, description in data_files.items():
        full_path = data_dir / file_path
        if not full_path.exists():
            warnings.append({
                "type": "missing_data",
                "path": file_path,
                "description": description,
                "impact": "Some features may not work"
            })
    
    # 3. 检查provider配置
    custom_dir = data_dir / "system/.secret/providers/custom"
    if custom_dir.exists():
        providers = list(custom_dir.glob("*.json"))
        if not providers:
            issues.append({
                "type": "no_provider",
                "path": "custom/",
                "description": "No AI provider configured",
                "fix": "Configure at least one provider in Settings → Providers"
            })
    
    return {
        "is_ok": len(issues) == 0,
        "issues": issues,
        "warnings": warnings
    }

def print_config_warning(issues: List[Dict], warnings: List[Dict]):
    """输出配置警告信息
    
    Args:
        issues: 问题列表
        warnings: 警告列表
    """
    if not issues and not warnings:
        return
    
    # 输出问题
    if issues:
        logger.warning("=" * 60)
        logger.warning("⚠️  System Configuration Issues Detected")
        logger.warning("=" * 60)
        
        for issue in issues:
            if issue["type"] == "missing_file":
                logger.warning(f"Missing: {issue['path']}")
                logger.warning(f"  {issue['description']}")
                logger.warning(f"  Fix: {issue['fix']}")
            elif issue["type"] == "no_provider":
                logger.warning(f"Warning: {issue['path']}")
                logger.warning(f"  {issue['description']}")
                logger.warning(f"  Fix: {issue['fix']}")
        
        logger.warning("")
        logger.warning("Quick diagnostics: python -m coapis.cli.doctor")
        logger.warning("=" * 60)
    
    # 输出警告
    if warnings:
        for warning in warnings:
            logger.warning(f"⚠️  {warning['path']}: {warning['description']}")
            logger.info(f"    Impact: {warning['impact']}")
```

**代码**：
```python
"""Default configuration generators."""
import json
from pathlib import Path
from typing import Dict, Any

def create_default_hidden_providers() -> Dict[str, Any]:
    """生成默认的隐藏provider列表"""
    return {
        "hidden": []  # 默认不隐藏任何provider
    }

def create_default_default_models() -> Dict[str, Any]:
    """生成默认的默认模型配置"""
    return {
        "chat": {
            "provider_id": "baidu",
            "model": "glm-5",
            "model_type": "chat"
        },
        "embedding": {
            "provider_id": "baidu",
            "model": "embedding-v1",
            "model_type": "embedding"
        },
        "rerank": None,
        "audio": None,
        "vision": None
    }

def create_default_active_model() -> Dict[str, Any]:
    """生成默认的激活模型配置"""
    return {
        "provider_id": "baidu",
        "model": "glm-5"
    }

def ensure_config_file(file_path: Path, generator_func, description: str) -> bool:
    """确保配置文件存在，不存在则创建默认配置
    
    Args:
        file_path: 配置文件路径
        generator_func: 生成默认配置的函数
        description: 配置描述
    
    Returns:
        True if file exists or created, False if failed
    """
    if file_path.exists():
        return True
    
    try:
        # 确保父目录存在
        file_path.parent.mkdir(parents=True, exist_ok=True)
        
        # 生成默认配置
        default_config = generator_func()
        
        # 写入文件
        with open(file_path, 'w', encoding='utf-8') as f:
            json.dump(default_config, f, indent=2, ensure_ascii=False)
        
        print(f"✅ Created default {description}: {file_path}")
        return True
    
    except Exception as e:
        print(f"❌ Failed to create {description}: {e}")
        return False

def ensure_all_configs(data_dir: Path):
    """确保所有必要配置文件存在
    
    Args:
        data_dir: 数据目录路径
    """
    configs = [
        (
            data_dir / "system/.secret/providers/hidden_providers.json",
            create_default_hidden_providers,
            "hidden providers config"
        ),
        (
            data_dir / "system/.secret/providers/default_models.json",
            create_default_default_models,
            "default models config"
        ),
        (
            data_dir / "system/.secret/providers/active_model.json",
            create_default_active_model,
            "active model config"
        ),
    ]
    
    for file_path, generator, description in configs:
        ensure_config_file(file_path, generator, description)
```
