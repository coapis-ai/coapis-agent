```typescript
import React, { useEffect, useState } from 'react';
import { List, Avatar, Empty } from 'antd';

import styles from './RecentWidget.module.less';

interface RecentWidgetProps {
  onStartScene: (sceneId: string) => void;
}

export const RecentWidget: React.FC<RecentWidgetProps> = ({
  onStartScene
}) => {
  const [recentScenes, setRecentScenes] = useState<RecentScene[]>([]);

  useEffect(() => {
    fetchRecentScenes();
  }, []);

  const fetchRecentScenes = async () => {
    const response = await api.get('/user-scenes/recent?limit=4');
    setRecentScenes(response.data.scenes);
  };

  const formatTime = (timestamp: string) => {
    const now = new Date();
    const time = new Date(timestamp);
    const diff = now.getTime() - time.getTime();
    
    const minutes = Math.floor(diff / 60000);
    const hours = Math.floor(diff / 3600000);
    const days = Math.floor(diff / 86400000);
    
    if (minutes < 60) return `${minutes}分钟前`;
    if (hours < 24) return `${hours}小时前`;
    if (days < 7) return `${days}天前`;
    return time.toLocaleDateString();
  };

  return (
    <div className={styles.recentWidget}>
      <h3>📅 最近使用</h3>
      
      {recentScenes.length === 0 ? (
        <Empty description="暂无使用记录" />
      ) : (
        <List
          dataSource={recentScenes}
          renderItem={(scene) => (
            <List.Item
              className={styles.listItem}
              onClick={() => onStartScene(scene.id)}
            >
              <List.Item.Meta
                avatar={<Avatar>{scene.icon}</Avatar>}
                title={scene.name}
                description={
                  <span>
                    <ClockCircleOutlined /> {formatTime(scene.last_used)}
                  </span>
                }
              />
            </List.Item>
          )}
        />
```

**Dashboard.tsx**：
```typescript
import React, { useState } from 'react';
import { Row, Col } from 'antd';
import { Widget } from './Widget';
import styles from './Dashboard.module.less';

interface DashboardProps {
  widgets: WidgetConfig[];
  onLayoutChange: (widgets: WidgetConfig[]) => void;
}

export const Dashboard: React.FC<DashboardProps> = ({
  widgets,
  onLayoutChange
}) => {
  const [draggingWidget, setDraggingWidget] = useState<string | null>(null);

  const handleDragStart = (e: React.DragEvent, widgetId: string) => {
    setDraggingWidget(widgetId);
    e.dataTransfer.effectAllowed = 'move';
  };

  const handleDragOver = (e: React.DragEvent) => {
    e.preventDefault();
    e.dataTransfer.dropEffect = 'move';
  };

  const handleDrop = (e: React.DragEvent, targetId: string) => {
    e.preventDefault();
    
    if (!draggingWidget || draggingWidget === targetId) return;

    const newWidgets = [...widgets];
    const dragIndex = newWidgets.findIndex(w => w.id === draggingWidget);
    const dropIndex = newWidgets.findIndex(w => w.id === targetId);

    // 交换位置
    [newWidgets[dragIndex].order, newWidgets[dropIndex].order] = 
      [newWidgets[dropIndex].order, newWidgets[dragIndex].order];

    onLayoutChange(newWidgets);
    setDraggingWidget(null);
  };

  const getWidgetSize = (size: string) => {
    switch (size) {
      case 'small': return 3;   // 25% width
      case 'medium': return 6;  // 50% width
      case 'large': return 12;  // 100% width
      default: return 6;
    }
  };

  return (
    <div className={styles.dashboard}>
      <Row gutter={[16, 16]}>
        {widgets
          .filter(w => w.visible)
          .sort((a, b) => a.order - b.order)
          .map(widget => (
            <Col 
              key={widget.id}
              span={getWidgetSize(widget.size)}
            >
              <Widget
                config={widget}
                onDragStart={(e) => handleDragStart(e, widget.id)}
                onDragOver={handleDragOver}
                onDrop={(e) => handleDrop(e, widget.id)}
              >
                {renderWidgetContent(widget)}
              </Widget>
            </Col>
          ))}
      </Row>
    </div>
  );
};

const renderWidgetContent = (widget: WidgetConfig) => {
  switch (widget.type) {
    case 'weather': return <WeatherWidget />;
    case 'welcome': return <WelcomeWidget />;
    case 'message': return <MessageWidget />;
    case 'recommend': return <RecommendWidget />;
    case 'recent': return <RecentWidget />;
    case 'favorite': return <FavoriteWidget />;
    case 'hot': return <HotWidget />;
    case 'stats': return <StatsWidget />;
    case 'quickActions': return <QuickActionsWidget />;
    default: return null;
  }
};
```

```typescript
import React from 'react';
import { Modal, List, Switch, Select, Button } from 'antd';
import styles from './WidgetSettings.module.less';

interface WidgetSettingsProps {
  visible: boolean;
  widgets: WidgetConfig[];
  onClose: () => void;
  onSave: (widgets: WidgetConfig[]) => void;
}

export const WidgetSettings: React.FC<WidgetSettingsProps> = ({
  visible,
  widgets,
  onClose,
  onSave
}) => {
  const [localWidgets, setLocalWidgets] = useState(widgets);

  const handleToggle = (widgetId: string, visible: boolean) => {
    setLocalWidgets(widgets.map(w => 
      w.id === widgetId ? { ...w, visible } : w
    ));
  };

  const handleSizeChange = (widgetId: string, size: string) => {
    setLocalWidgets(widgets.map(w =>
      w.id === widgetId ? { ...w, size } : w
    ));
  };

  const handleSave = () => {
    onSave(localWidgets);
    onClose();
  };

  return (
    <Modal
      title="自定义首页"
      open={visible}
      onCancel={onClose}
      footer={[
        <Button key="cancel" onClick={onClose}>取消</Button>,
        <Button key="save" type="primary" onClick={handleSave}>保存</Button>
      ]}
    >
      <List
        dataSource={localWidgets}
        renderItem={(widget) => (
          <List.Item>
            <div className={styles.widgetSetting}>
              <div className={styles.info}>
                <Switch
                  checked={widget.visible}
```

#### 任务3.3：实现布局持久化（0.3天）

```typescript
// hooks/useWidgetLayout.ts
import { useState, useEffect } from 'react';

export const useWidgetLayout = () => {
  const [widgets, setWidgets] = useState<WidgetConfig[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadLayout();
  }, []);

  const loadLayout = async () => {
    try {
      const response = await api.get('/user/widget-layout');
      setWidgets(response.data.widgets);
    } catch (error) {
      // 使用默认布局
      setWidgets(getDefaultLayout());
    } finally {
      setLoading(false);
    }
  };

  const saveLayout = async (newWidgets: WidgetConfig[]) => {
    try {
      await api.post('/user/widget-layout', { widgets: newWidgets });
      setWidgets(newWidgets);
    } catch (error) {
      console.error('Failed to save layout:', error);
    }
  };

  return { widgets, loading, saveLayout };
};

const getDefaultLayout = (): WidgetConfig[] => {
  return [
```

## 数据库设计

### 用户偏好表

```sql
CREATE TABLE user_preferences (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id VARCHAR(255) NOT NULL UNIQUE,
    function_tags TEXT,  -- JSON array: ["国土资源", "环境保护"]
    favorite_scenes TEXT,  -- JSON array: ["scene_id_1", "scene_id_2"]
    widget_layout TEXT,  -- JSON array of widget configs
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(username)
);
```

### 场景使用记录表

```sql
CREATE TABLE scene_usage (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id VARCHAR(255) NOT NULL,
    scene_id VARCHAR(255) NOT NULL,
    used_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(username),
    FOREIGN KEY (scene_id) REFERENCES scenes(id)
);

CREATE INDEX idx_scene_usage_user ON scene_usage(user_id, used_at DESC);
```

### 消息通知表

```sql
CREATE TABLE messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id VARCHAR(255) NOT NULL,
    type VARCHAR(50) NOT NULL,  -- system, task, scene
    title VARCHAR(255) NOT NULL,
    content TEXT,
    read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(username)
);

CREATE INDEX idx_messages_user ON messages(user_id, created_at DESC);
```

## 风险评估

### 技术风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 天气API不稳定 | 天气卡片无法显示 | 中 | 使用缓存，失败时显示默认数据 |
| 拖拽功能在移动端不友好 | 用户体验差 | 中 | 移动端禁用拖拽，使用上下箭头调整顺序 |
| 推荐算法不准确 | 推荐场景不相关 | 低 | 提供手动调整，记录用户反馈 |
| 卡片数量过多导致性能问题 | 页面卡顿 | 低 | 限制最大卡片数量，支持分页 |

### 进度风险

| 风险 | 影响 | 概率 | 应对措施 |
|------|------|------|----------|
| 后端API开发延期 | 前端无法测试 | 中 | 先用mock数据开发前端，后端并行开发 |
| 天气API接入困难 | 天气功能无法上线 | 低 | 作为可选功能，后期补充 |
| 数据库迁移问题 | 现有数据丢失 | 低 | 提前备份，逐步迁移 |
