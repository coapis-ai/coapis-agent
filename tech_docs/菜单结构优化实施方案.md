```typescript
import { MessageOutlined } from '@ant-design/icons';
import ChatWindow from './ChatWindow';
import styles from './styles.module.css';

export default function FloatingChat() {
  const [isOpen, setIsOpen] = useState(false);
  const [unreadCount, setUnreadCount] = useState(0);
  
  const toggleChat = () => {
    setIsOpen(!isOpen);
    if (!isOpen) {
      setUnreadCount(0);  // 打开时清空未读数

  };
  
  return (
    <div className={styles.floatingChat}>
      {isOpen && <ChatWindow onClose={() => setIsOpen(false)} />}
      
      <div className={styles.chatButton} onClick={toggleChat}>
        <Badge count={unreadCount} size="small">
          <div className={styles.buttonIcon}>
            <MessageOutlined style={{ fontSize: 24 }} />
          </div>
        </Badge>
      </div>
    </div>
  );
}
```

```typescript
  const handleSend = () => {

    setMessages([...messages, { role: 'user', content: inputValue }]);
    
    // 发送到AI（TODO）
    // ...
    
    setInputValue('');
  };
  
  return (
    <div className={styles.chatWindow}>
      <div className={styles.windowHeader}>
        <span>AI助手</span>
        <CloseOutlined onClick={onClose} style={{ cursor: 'pointer' }} />
```

**服务层**：

```python
# server/coapis/services/user_tag_service.py

from ..database import get_db

class UserTagService:
    def __init__(self):
        self.db = get_db()
    
    async def get_user_tags(self, user_id: str):
        """获取用户的专业标签"""
        user_tags = await self.db.user_tags.find({"user_id": user_id}).to_list(None)
        tag_ids = [ut["tag_id"] for ut in user_tags]
        
        tags = await self.db.tags.find({"id": {"$in": tag_ids}}).to_list(None)
        return tags
    
    async def set_user_tags(self, user_id: str, tag_ids: list[str]):
        """设置用户的专业标签"""
        # 删除旧的
        await self.db.user_tags.delete_many({"user_id": user_id})
        
        # 添加新的
        for tag_id in tag_ids:
            await self.db.user_tags.insert_one({
                "user_id": user_id,
                "tag_id": tag_id
            })
    
    async def get_all_tags(self):
        """获取所有标签（按分类分组）"""
        tags = await self.db.tags.find().to_list(None)
        
        # 按分类分组
        grouped = {
            "business": [],
            "industry": [],
            "tech": []
        }
        
        for tag in tags:
            category = tag.get("category", "business")
            if category in grouped:
                grouped[category].append(tag)
        
        return grouped
    
    async def get_recommended_scenes(self, user_id: str):
        """根据用户标签推荐场景"""
        # 1. 获取用户标签
        user_tags = await self.get_user_tags(user_id)
        tag_ids = [t["id"] for t in user_tags]
        
        # 2. 查找匹配标签的场景
        scenes = await self.db.scenes.find({
            "tag_ids": {"$in": tag_ids}
        }).sort("usage_count", -1).limit(10).to_list(None)
        
        return scenes
```

```typescript
import { useState } from 'react';
import { Button, Card, Tag } from 'antd';
import { EditOutlined } from '@ant-design/icons';
import TagSelector from './TagSelector';
import RecommendedScenes from './RecommendedScenes';
import { useQuery } from '@tanstack/react-query';
import { api } from '@/api';
import styles from './styles.module.css';

export default function MyDomain() {
  const [showTagSelector, setShowTagSelector] = useState(false);
  
  const { data: userTags = [] } = useQuery({
    queryKey: ['user-tags'],
    queryFn: () => api.get('/user-tags'),
  });
  
  const { data: recommendedScenes = [] } = useQuery({
    queryKey: ['recommended-scenes'],
    queryFn: () => api.get('/scenes/recommended'),
  });
  
  return (
    <div className={styles.myDomain}>
      <h1>我的领域</h1>
      
      {/* 我的专业标签 */}
      <Card 
        title="🏷️ 我的专业标签"
        extra={
          <Button 
            type="link" 
            icon={<EditOutlined />}
            onClick={() => setShowTagSelector(true)}
          >
            编辑标签
          </Button>
        }
      >
        <div className={styles.tagList}>
          {userTags.length === 0 ? (
            <div className={styles.emptyTag}>
              <p>您还没有选择专业标签</p>
              <Button type="primary" onClick={() => setShowTagSelector(true)}>
                选择标签
              </Button>
            </div>
          ) : (
            userTags.map(tag => (
              <Tag key={tag.id} color="blue" className={styles.tagItem}>
                {tag.icon} {tag.name}
              </Tag>
            ))
          )}
        </div>
      </Card>
      
      {/* 推荐场景 */}
      <RecommendedScenes scenes={recommendedScenes} />
      
      {/* 标签选择器弹窗 */}
      {showTagSelector && (
        <TagSelector
          selectedTags={userTags}
          onClose={() => setShowTagSelector(false)}
          onSave={(tags) => {
            // 保存标签
            api.post('/user-tags', tags);
            setShowTagSelector(false);
          }}
        />
      )}
    </div>
  );
}
```

```typescript
// client/src/pages/Workbench/MyDomain/TagSelector.tsx

import { useState, useEffect } from 'react';
import { Modal, Checkbox, Button } from 'antd';
import { useQuery } from '@tanstack/react-query';
import { api } from '@/api';
import styles from './styles.module.css';

export default function TagSelector({ selectedTags, onClose, onSave }) {
  const [checkedTags, setCheckedTags] = useState<string[]>(
    selectedTags.map(t => t.id)
  );
  
  const { data: allTags = {} } = useQuery({
    queryKey: ['all-tags'],
    queryFn: () => api.get('/tags'),
  });
  
  const handleSave = () => {
    onSave(checkedTags);
  };
  
  return (
    <Modal
      title="选择您的专业领域"
      open={true}
      onCancel={onClose}
      footer={[
        <Button key="cancel" onClick={onClose}>取消</Button>,
        <Button key="save" type="primary" onClick={handleSave}>保存</Button>,
      ]}
      width={600}
    >
      <div className={styles.tagSelector}>
        {/* 业务职能 */}
        <div className={styles.tagCategory}>
          <h3>💼 业务职能</h3>
          <div className={styles.tagGrid}>
            {allTags.business?.map(tag => (
              <Checkbox
                key={tag.id}
                checked={checkedTags.includes(tag.id)}
                onChange={(e) => {
                  if (e.target.checked) {
                    setCheckedTags([...checkedTags, tag.id]);
                  } else {
                    setCheckedTags(checkedTags.filter(id => id !== tag.id));

                }}

                {tag.icon} {tag.name}
              </Checkbox>
            ))}
```
