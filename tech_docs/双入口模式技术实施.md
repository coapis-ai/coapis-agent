## 一、技术架构设计

### 1.1 整体架构

```
┌─────────────────────────────────────────────────────────┐
│  前端层 (React + TypeScript)                            │
│  ├─ WorkbenchLayout (工作台布局)                        │
│  ├─ ConsoleLayout (控制台布局)                          │
│  ├─ PermissionGuard (权限守卫)                          │
│  └─ SharedComponents (共享组件)                         │
└─────────────────────────────────────────────────────────┘
                          ↓ REST API
┌─────────────────────────────────────────────────────────┐
│  后端层 (FastAPI + Python)                              │
│  ├─ AuthMiddleware (认证中间件)                         │
│  ├─ PermissionMiddleware (权限中间件)                   │
│  ├─ RouteHandler (路由处理)                             │
│  └─ ServiceLayer (业务逻辑)                             │
└─────────────────────────────────────────────────────────┘
                          ↓
┌─────────────────────────────────────────────────────────┐
│  数据层 (SQLite / PostgreSQL)                           │
│  ├─ users (用户表)                                      │
│  ├─ roles (角色表)                                      │
│  ├─ permissions (权限表)                                │
│  └─ role_permissions (角色权限关联表)                   │
└─────────────────────────────────────────────────────────┘
```

## 二、数据库设计

### 2.1 权限系统数据模型

```sql
-- 用户表（扩展现有）
ALTER TABLE users ADD COLUMN preferred_entry VARCHAR(20) DEFAULT 'workbench';

-- 角色表
CREATE TABLE roles (
    id VARCHAR(50) PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    level INTEGER DEFAULT 0,
    is_default BOOLEAN DEFAULT FALSE,
    description TEXT
);

-- 权限表
CREATE TABLE permissions (
    id VARCHAR(50) PRIMARY KEY,
    module VARCHAR(50) NOT NULL,
    action VARCHAR(20) NOT NULL,
    description TEXT,
    UNIQUE(module, action)
);

-- 角色权限关联表
CREATE TABLE role_permissions (
    role_id VARCHAR(50) REFERENCES roles(id),
    permission_id VARCHAR(50) REFERENCES permissions(id),
    scope VARCHAR(20) DEFAULT 'all',
    PRIMARY KEY (role_id, permission_id)
);
```

## 四、权限系统实现

### 4.1 权限装饰器

```python
def require_permission(module: str, action: str):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, user: User = None, **kwargs):
            if not user:
                raise HTTPException(401, "Unauthorized")
            
            if user.role == "super_admin":
                return await func(*args, user=user, **kwargs)
            
            has_permission = await check_permission(user.id, module, action)
            if not has_permission:
                raise HTTPException(403, f"No {action} permission for {module}")
            
            return await func(*args, user=user, **kwargs)
        return wrapper
    return decorator
```

### 4.2 权限服务

```python
class PermissionService:
    async def check_permission(self, user_id: str, module: str, action: str) -> bool:
        user = await self.db.users.find_one({"id": user_id})
        if user["role"] == "super_admin":
            return True
        
        permissions = await self.get_user_permissions(user_id)
        return any(
            p["module"] == module and p["action"] in [action, "manage"]
            for p in permissions
        )
```

## 五、前端实现

### 5.1 权限Hook

```typescript
export function usePermission() {
  const { data: permissions = [] } = useQuery({
    queryKey: ['permissions'],
    queryFn: () => api.get('/auth/permissions'),
  });
  
  const hasPermission = (module: string, action: string): boolean => {
    return permissions.some(
      p => p.module === module && (p.action === action || p.action === 'manage')
    );
  };
  
  return { hasPermission, permissions };
}
```

### 5.2 权限守卫

```typescript
export function PermissionGuard({ module, action, children, fallback }) {
  const { hasPermission } = usePermission();
  
  if (!hasPermission(module, action)) {
    return <>{fallback}</>;
  }
  
  return <>{children}</>;
}
```
